public static async Task Start(int i)
{
    for (int j = 0; j < 100; j++)
    {
        Console.WriteLine("thread " + i + " " + j);
    }

    // Async delay without blocking
    await Task.Delay(1000);  // Await the delay to simulate async work (like waiting for data)
}

static async Task Main(string[] args)
{
    // Start the asynchronous tasks
    var x = Start(1);
    var y = Start(2);
    var z = Start(3);

    // Wait for all tasks to finish
    await Task.WhenAll(x, y, z);
}
